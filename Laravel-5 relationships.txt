class Post extends Model
{
    use HasFactory;
    protected $fillable =[
        'user_id',
        'title',
        'description',
        'thumbnail'

    ];

    public function user()
    {
        return $this->belongsTo(related: User::class, foreignKey:'user_id');
    }
}

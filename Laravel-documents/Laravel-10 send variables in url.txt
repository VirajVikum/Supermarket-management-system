Route::get('/dashboard', [App\Http\Controllers\HomeController::class, 'dashboard'])->name('dashboard');
Route::get('/leads', [App\Http\Controllers\HomeController::class, 'leads'])->name('leads');





public function dashboard(Request $request)
    {
        $variable = $request->query('var');
        view()->share('variable', $variable);  //To share with extended views
        return view('admin/dashboard')->with('variable', $variable);
    }

    public function leads(Request $request)
    {
        $variable = $request->query('var');
        view()->share('variable', $variable);  //To share with extended views
        return view('admin/leads')->with('variable', $variable);
    }

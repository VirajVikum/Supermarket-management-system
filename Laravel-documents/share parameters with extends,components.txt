public function dashboard(Request $request)
    {
        

        $agents = ac_user::all();
        $variable = "Dashboard";
        $status=1;

        view()->share([
            'agents'  => $agents,
            'variable' => $variable,
            'status'   => $status,
        ]);

       

        
        return view('admin.dashboard', compact('agents', 'variable','status'));
       }

class LoginClient extends Component
{
    public $email;
    public $password;

    public function login()
    {
        $email=$this->email;
        $password=$this->password;
        // dd('$email');

        $item = Client::where('email', $email)->first();
        // dd('$item->password');

        if($item)
        {
            if($password===$item->password)
            {
                // dd('$item->password');
                return redirect()->to(route('home'));
            }

        }
    }








Route::get('/login',LoginClient::class)->name('login');
